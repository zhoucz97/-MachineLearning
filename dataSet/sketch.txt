man = []
other = []
try:
	data = open('zhou.txt')
	
	for each_line in data:
		try:
			(role, line_spoken) = each_line.split(':', 1)
			line_spoken = line_spoken.strip()
			if role == 'A':
				man.append(line_spoken)
			elif role == 'B':
				other.append(line_spoken)
		except ValueError:
			pass
		
	data.close()
except FileNotFoundError:
	print('The datafile is missing!')
	
print(man)
print(other)
